45 inline matrix_cl cholesky_decompose(matrix_cl& A) {
46  if (A.rows() == 0)
47  return A;
48  // Repeats the blocked cholesky decomposition until the size of the remaining
49  // submatrix is smaller or equal to the minimum blocks size
50  // or a heuristic of 100.
51  // The Cholesky (OpenCL) algorithm only uses one local block so we need the
52  // matrix To be less than the max thread block size.
53  if (A.rows() <= opencl_context.tuning_opts().cholesky_min_L11_size) {
54  matrix_cl L(A.rows(), A.cols());
55  try {
56  opencl_kernels::cholesky_decompose(cl::NDRange(A.rows()),
57  cl::NDRange(A.rows()), A.buffer(),
58  L.buffer(), A.rows());
59  } catch (const cl::Error& e) {
60  check_opencl_error("cholesky_decompose", e);
61  }
62  return L;
63  }
64  // NOTE: The code in this section follows the naming conventions
65  // in the report linked in the docs.
66  const int block
67  = std::floor(A.rows() / opencl_context.tuning_opts().cholesky_partition);
68  // Subset the top left block of the input A into A_11
69  matrix_cl A_11(block, block);
70  A_11.sub_block(A, 0, 0, 0, 0, block, block);
71  // The following function either calls the
72  // blocked cholesky recursively for the submatrix A_11
73  // or calls the kernel directly if the size of the block is small enough
74  matrix_cl L_11 = cholesky_decompose(A_11);
75  // Copies L_11 back to the input matrix
76  A.sub_block(L_11, 0, 0, 0, 0, block, block);
77 
78  const int block_subset = A.rows() - block;
79  matrix_cl A_21(block_subset, block);
80  A_21.sub_block(A, block, 0, 0, 0, block_subset, block);
81  // computes A_21*((L_11^-1)^T)
82  // and copies the resulting submatrix to the lower left hand corner of A
83  matrix_cl L_21
84  = opencl::multiply<TriangularViewCL::Entire, TriangularViewCL::Upper>(
85  A_21, transpose(lower_triangular_inverse(L_11)));
86  A.sub_block(L_21, 0, 0, block, 0, block_subset, block);
87  matrix_cl A_22(block_subset, block_subset);
88  A_22.sub_block(A, block, block, 0, 0, block_subset, block_subset);
89  // computes A_22 - L_21*(L_21^T)
90  matrix_cl L_22 = A_22 - multiply_transpose(L_21);
