270  const int num_jobs = sum(job_chunks);
271 
272  const int first_job
273  = std::accumulate(job_chunks.begin(), job_chunks.begin() + rank_, 0);
274  const int num_local_jobs = local_job_params_dbl_.cols();
275  int local_outputs_per_job = num_local_jobs == 0 ? 0 : num_outputs_per_job_;

278  num_local_jobs);

280 
281  typename cache_x_r::cache_t& local_x_r = cache_x_r::data();
282  typename cache_x_i::cache_t& local_x_i = cache_x_i::data();
283 
284  // check if we know already output sizes
285  if (cache_f_out::is_valid()) {
286  typename cache_f_out::cache_t& f_out = cache_f_out::data();
287  const int num_outputs
288  = std::accumulate(f_out.begin() + first_job,
289  f_out.begin() + first_job + num_local_jobs, 0);
290  local_output.resize(Eigen::NoChange, num_outputs);
291  }
292 
293  int local_ok = 1;
294  try {
295  for (int i = 0, offset = 0; i < num_local_jobs;
296  offset += local_f_out[i], ++i) {
297  const matrix_d job_output
298  = ReduceF()(local_shared_params_dbl_, local_job_params_dbl_.col(i),
299  local_x_r[i], local_x_i[i], 0);
300  local_f_out[i] = job_output.cols();
301 
302  if (local_outputs_per_job == -1) {
303  local_outputs_per_job = job_output.rows();
304  // changing rows, but leave column size as is
305  local_output.conservativeResize(local_outputs_per_job,
306  Eigen::NoChange);
307  }
308 
309  if (local_output.cols() < offset + local_f_out[i])
310  // leave row size, but change columns size
311  local_output.conservativeResize(Eigen::NoChange,
312  2 * (offset + local_f_out[i]));
313 
314  local_output.block(0, offset, local_output.rows(), local_f_out[i])
315  = job_output;
316  }
317  } catch (const std::exception& e) {
318  // see note 1 above for an explanation why we do not rethrow
319  // here, but mereley flag it to keep the cluster synchronized
320  local_ok = 0;
321  }
322 
323  // during first execution we distribute the output sizes from
324  // local jobs to the root. This needs to be done for the number of
325  // outputs of each result and the number of outputs per job.
326  if (num_outputs_per_job_ == -1) {

328  // that no exception has been fired from any node. Hence,
329  // share the info about the current status across all nodes.

332  bool all_ok = cluster_status == static_cast<int>(world_size_);

334  if (!all_ok) {

336  if (rank_ == 0) {
337  throw std::domain_error("MPI error on first evaluation.");
338  }

340  // listening state
341  return result_t();
342  }
343 
344  // execution was ok everywhere such that we can now distribute the
345  // outputs per job as recorded on the root on the first execution
346  boost::mpi::broadcast(world_, local_outputs_per_job, 0);
347  num_outputs_per_job_ = local_outputs_per_job;

350  // now given that this first execution was just fine
351  if (!cache_f_out::is_valid()) {

353  boost::mpi::gatherv(world_, local_f_out.data(), num_local_jobs,
354  world_f_out.data(), job_chunks, 0);
355  // on the root we now have all sizes from all children. Copy
356  // over the local sizes to the world vector on each local node
357  // in order to cache this information locally.
358  std::copy(local_f_out.begin(), local_f_out.end(),
359  world_f_out.begin() + first_job);
