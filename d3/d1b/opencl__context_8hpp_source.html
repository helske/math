104  // older integrated GPUs or CPUs
105  if (thread_block_size_sqrt < base_opts_["THREAD_BLOCK_SIZE"]) {
106  base_opts_["THREAD_BLOCK_SIZE"] = thread_block_size_sqrt;
107  base_opts_["WORK_PER_THREAD"] = 1;
108  }
109  // Thread block size for the Cholesky
110  // TODO(Steve): This should be tuned in a higher part of the stan language
111  if (max_thread_block_size_ >= 256) {
