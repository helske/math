34  std::map<const char*, int> options) {
35  std::string kernel_opts = "";
36  for (auto&& comp_opts : options) {
37  kernel_opts += std::string(" -D") + comp_opts.first + "="
38  + std::to_string(comp_opts.second);
39  }
40  std::string kernel_source(opencl_kernels::helpers);
41  kernel_source.append(source);
42  cl::Program program;
43  try {
44  cl::Program::Sources src(1, std::make_pair(kernel_source.c_str(),
45  strlen(kernel_source.c_str())));
46  program = cl::Program(opencl_context.context(), src);
47  program.build({opencl_context.device()}, kernel_opts.c_str());
48 
49  return cl::Kernel(program, name);
50  } catch (const cl::Error& e) {
51  // in case of CL_BUILD_PROGRAM_FAILURE, print the build error
52  if (e.err() == -11) {

54  opencl_context.device()[0]);

56  } else {
57  check_opencl_error(name, e);
58  }
59  }

61 }
